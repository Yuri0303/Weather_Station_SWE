public class HelloWorld {

    public static int somma(int a, int b) {
        return a + b;
    }

    public static void main(String[] args) {

        int x = 10;
        int y = 20;

        int risultato = somma(x, y);

        System.out.println("Risultato = " + risultato);

        for (int i = 0; i < 5; i++) {
            System.out.println("Iterazione " + i);
        }
    }
}
